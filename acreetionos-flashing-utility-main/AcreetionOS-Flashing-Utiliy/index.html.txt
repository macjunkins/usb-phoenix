    <script>
        // Create floating particles
        function createParticles() {
            const container = document.querySelector('.floating-particles');
            for (let i = 0; i < 20; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.animationDelay = Math.random() * 20 + 's';
                particle.style.animationDuration = (20 + Math.random() * 10) + 's';
                container.appendChild(particle);
            }
        }

        // Update status with appropriate styling
        function updateStatus(message, type = 'info') {
            const statusEl = document.getElementById('status');
            const progressBar = document.getElementById('progress-bar');
            
            let icon = 'üíæ';
            if (message.includes('‚¨áÔ∏è') || message.includes('Downloading')) {
                icon = '‚¨áÔ∏è';
                type = 'loading';
            } else if (message.includes('üî•') || message.includes('Flashing')) {
                icon = 'üî•';
                type = 'loading';
            } else if (message.includes('‚úÖ') || message.includes('complete')) {
                icon = '‚úÖ';
                type = 'success';
            } else if (message.includes('‚ùå') || message.includes('failed') || message.includes('canceled')) {
                icon = '‚ùå';
                type = 'error';
            }

            statusEl.innerHTML = `<span class="status-icon">${icon}</span>${message.replace(/[‚¨áÔ∏èüî•‚úÖ‚ùå]/g, '').trim()}`;
            
            if (type === 'loading') {
                statusEl.innerHTML += '<div class="loading-spinner"></div>';
                progressBar.style.display = 'block';
            } else {
                progressBar.style.display = 'none';
            }
        }

        function updateProgress(percent) {
            const progressFill = document.getElementById('progress-fill');
            progressFill.style.width = percent + '%';
        }

        // Initialize particles when page loads
        document.addEventListener('DOMContentLoaded', () => {
            createParticles();
            loadISOs();
            loadUSBDevices();
        });

        // Add some interactive effects
        document.querySelectorAll('.btn').forEach(btn => {
            btn.addEventListener('mouseenter', () => {
                if (!btn.disabled) {
                    btn.style.transform = 'translateY(-2px)';
                }
            });
            
            btn.addEventListener('mouseleave', () => {
                btn.style.transform = 'translateY(0)';
            });
        });
    </script>
